\section{Basics} \hfill \break

\noindent 1. Power Rule
$$ \frac{d}{dx}[x^n] = nx^{n-1}$$

\noindent 2. Constant Rule
 $$ \frac{d}{dx}[a] = 0 \ $$

\noindent 3. Sum and Difference Rule
$$ \frac{d}{dx}[f(x) \pm g(x)] = \frac{d}{dx}[f(x)] \pm \frac{d}{dx}[g(x)] \ $$

\noindent 4. Product Rule
$$  \frac{d}{dx}[f(x)g(x)] = f'(x)g(x) + f(x)g'(x) \ $$ \\

\section{Trigonometric Derivatives} \hfill \break

\noindent 1. Sine Rule
$$  \frac{d}{dx}[\sin(x)] = \cos(x) \ $$

\noindent 2. Cosine Rule
 $$ \frac{d}{dx}[\cos(x)] = -\sin(x) \ $$

\noindent 3. Tangent Rule
$$ \frac{d}{dx}[\tan(x)] = \sec^2(x) \ $$

\noindent 4. Cosecant Rule
$$   \frac{d}{dx}[\csc(x)] = -\csc(x) \cot(x) \ $$

\noindent 5. Secant Rule
$$ \frac{d}{dx}[\sec(x)] = \sec(x) \tan(x) \ $$

\break



\section{Exponential and Logarithmic Derivatives} \hfill \break

\noindent 1. Exponential Rule
$$ \frac{d}{dx}[e^x] = e^x \ $$

\noindent 2. Natural Logarithm Rule
 $$  \frac{d}{dx}[\ln(x)] = \frac{1}{x} \ $$

\noindent 3. General Logarithm Rule
$$  \frac{d}{dx}[\log_a(x)] = \frac{1}{x \ln(a)} \ $$

\noindent 4. Exponential Function Rule
$$  \frac{d}{dx}[a^x] = a^x \ln(a) \ $$

\noindent 5. Logarithmic Function Rule
$$  \frac{d}{dx}[\log_a(x)] = \frac{1}{x \ln(a)} \ $$ \\
